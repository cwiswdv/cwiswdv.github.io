using System;
using static System.Console;
namespace if_else
{
    internal class Program
    {
        static void Main(string[] args)
        {
            int num1, num2, num3;

            Write("Enter first integer: ");
            num1 = Convert.ToInt32(ReadLine());

            Write("Enter second integer: ");
            num2 = Convert.ToInt32(ReadLine());

            Write("Enter third integer: ");
            num3 = Convert.ToInt32(ReadLine());

            if (num1 == num2)
                if (num3 == num1)
                        WriteLine("All three numbers are equal");
                else
                        WriteLine("First two are equal");
                else if (num1 == num3)
                        WriteLine("First and last are equal");
                else if (num2 == num3)
                        WriteLine("Last two are equal");
                else
                        WriteLine("No two numbers are equal");

        }
    }
}
